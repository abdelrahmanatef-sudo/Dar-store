    <script>

        let cart = JSON.parse(localStorage.getItem('darCart')) || [];

 

        function updateCartCount() {

            const count = cart.reduce((sum, item) => sum + item.qty, 0);

            document.getElementById('cartCount').textContent = count;

            document.getElementById('itemsCount').textContent = `(${count} منتج)`;

        }

 

        function renderCart() {

            const container = document.getElementById('cartContainer');

            const summary = document.getElementById('cartSummary');

 

            if (cart.length === 0) {

                container.innerHTML = `

                    <div class="empty-cart">

                        <i class="fas fa-shopping-bag"></i>

                        <h2>سلة التسوق فارغة</h2>

                        <p>لم تضف أي منتجات إلى سلتك بعد. استكشف منتجاتنا وابدأ التسوق!</p>

                        <a href="index.html" class="shop-btn">

                            <i class="fas fa-shopping-bag"></i>

                            ابدأ التسوق

                        </a>

                    </div>

                `;

                summary.style.display = 'none';

                return;

            }

 

            summary.style.display = 'block';

            container.innerHTML = cart.map((item, index) => `

                <div class="cart-item">

                    <img src="${item.image}" alt="${item.name}" class="cart-item-img">

                    <div class="cart-item-info">

                        <div class="cart-item-name">${item.name}</div>

                        <div class="cart-item-price">${item.price} ج.م / قطعة</div>

                    </div>

                    <div class="cart-item-qty">

                        <button class="qty-btn" onclick="updateQty(${index}, -1)">-</button>

                        <span class="qty-value">${item.qty}</span>

                        <button class="qty-btn" onclick="updateQty(${index}, 1)">+</button>

                    </div>

                    <div class="cart-item-total">${item.price * item.qty} ج.م</div>

                    <button class="remove-btn" onclick="removeItem(${index})">

                        <i class="fas fa-trash-alt"></i>

                    </button>

                </div>

            `).join('');

 

            updateTotals();

        }

 

        function updateQty(index, change) {

            cart[index].qty += change;

            if (cart[index].qty <= 0) {

                cart.splice(index, 1);

            }

            localStorage.setItem('darCart', JSON.stringify(cart));

            renderCart();

            updateCartCount();

        }

 

        function removeItem(index) {

            cart.splice(index, 1);

            localStorage.setItem('darCart', JSON.stringify(cart));

            renderCart();

            updateCartCount();

        }

 

        function updateTotals() {

            const subtotal = cart.reduce((sum, item) => sum + (item.price * item.qty), 0);

            const shipping = subtotal > 500 ? 0 : 50;

            const discount = 0; // Will be updated by promo code

            const total = subtotal + shipping - discount;

 

            document.getElementById('subtotal').textContent = subtotal.toLocaleString() + ' ج.م';

            document.getElementById('shipping').textContent = shipping === 0 ? 'مجاني' : shipping + ' ج.م';

            document.getElementById('discount').textContent = discount > 0 ? '-' + discount.toLocaleString() + ' ج.م' : '0 ج.م';

            document.getElementById('total').textContent = total.toLocaleString() + ' ج.م';

        }

 

        function applyPromo() {

            const code = document.getElementById('promoInput').value.trim();

            if (code === 'DAR10') {

                const subtotal = cart.reduce((sum, item) => sum + (item.price * item.qty), 0);

                const discount = Math.round(subtotal * 0.1);

                document.getElementById('discount').textContent = '-' + discount.toLocaleString() + ' ج.م';

                const shipping = subtotal > 500 ? 0 : 50;

                const total = subtotal + shipping - discount;

                document.getElementById('total').textContent = total.toLocaleString() + ' ج.م';

                alert('تم تطبيق خصم 10%!');

            } else {

                alert('كود الخصم غير صالح');

            }

        }

 

        // Initialize

        renderCart();

        updateCartCount();

    </script>
