# Update login.html to have proper links and header

with open('/mnt/agents/upload/login.html', 'r', encoding='utf-8') as f:

    login_content = f.read()

 

# Add a link back to store in the login page

login_updated = login_content.replace(

    '<div class="auth-logo">',

    '<a href="index.html" style="text-decoration: none; display: block; margin-bottom: 1rem;"><div class="auth-logo">'

).replace(

    '</div>\n\n            <div class="tabs">',

    '</div></a>\n\n            <div class="tabs">'

)

 

# Update handleLogin to redirect properly

login_updated = login_updated.replace(

    "window.location.href = 'index.html';",

    "window.location.href = 'index.html';"

)

 

with open('/mnt/agents/output/login.html', 'w', encoding='utf-8') as f:

    f.write(login_updated)

 

print("✅ login.html updated successfully!")
